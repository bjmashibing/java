018package org.apache.commons.dbcp2.datasources;
019
020import javax.naming.RefAddr;
021import javax.naming.Reference;
022
023/**
024 * A JNDI ObjectFactory which creates <code>SharedPoolDataSource</code>s
025 *
026 * @since 2.0
027 */
028public class SharedPoolDataSourceFactory extends InstanceKeyDataSourceFactory {
029    private static final String SHARED_POOL_CLASSNAME = SharedPoolDataSource.class.getName();
030
031    @Override
032    protected boolean isCorrectClass(final String className) {
033        return SHARED_POOL_CLASSNAME.equals(className);
034    }
035
036    @Override
037    protected InstanceKeyDataSource getNewInstance(final Reference ref) {
038        final SharedPoolDataSource spds = new SharedPoolDataSource();
039        final RefAddr ra = ref.get("maxTotal");
040        if (ra != null && ra.getContent() != null) {
041            spds.setMaxTotal(Integer.parseInt(ra.getContent().toString()));
042        }
043        return spds;
044    }
045}
