017package org.apache.commons.dbcp2;
018
019import org.apache.commons.logging.Log;
020import org.apache.commons.pool2.SwallowedExceptionListener;
021
022/**
023 * Class for logging swallowed exceptions.
024 *
025 * @since 2.0
026 */
027public class SwallowedExceptionLogger implements SwallowedExceptionListener {
028
029    private final Log log;
030    private final boolean logExpiredConnections;
031
032    /**
033     * Create a SwallowedExceptionLogger with the given logger. By default, expired connection logging is turned on.
034     *
035     * @param log
036     *            logger
037     */
038    public SwallowedExceptionLogger(final Log log) {
039        this(log, true);
040    }
041
042    /**
043     * Create a SwallowedExceptionLogger with the given logger and expired connection logging property.
044     *
045     * @param log
046     *            logger
047     * @param logExpiredConnections
048     *            false suppresses logging of expired connection events
049     */
050    public SwallowedExceptionLogger(final Log log, final boolean logExpiredConnections) {
051        this.log = log;
052        this.logExpiredConnections = logExpiredConnections;
053    }
054
055    @Override
056    public void onSwallowException(final Exception e) {
057        if (logExpiredConnections || !(e instanceof LifetimeExceededException)) {
058            log.warn(Utils.getMessage("swallowedExceptionLogger.onSwallowedException"), e);
059        }
060    }
061}
